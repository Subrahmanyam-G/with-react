Introduction

JavaScript is a scripting language. A scripting language is a lightweight programming language.
  • JavaScript code can be inserted into HTML pages.
  • It can be executed by all modern web browsers.

The <script> Tag

Step 1: Add a script tag inside your HTML.
<script type="text/javascript">
// JavaScript code here
</script>
✅ You can place scripts in both the head and body sections.

Output in JavaScript

1) document.write()

Writes text directly into the page body.

<html>
<head>
  <script type="text/javascript">
    document.write("Hello... Welcome to JavaScript Class from Reactify Hub");
  </script>
</head>
<body></body>
</html>

2) innerHTML

Inserts text inside an HTML element.

<html>
<head></head>
<body>
  <h1 id="special"></h1>
  <script type="text/javascript">
    document.getElementById("special").innerHTML = "Welcome to JS Class from Reactify Hub";
  </script>
</body>
</html>

3) window.alert()

Displays an alert box with a message.

<html>
<head></head>
<body>
  <script type="text/javascript">
    window.alert("Welcome to JS from Reactify Hub");
  </script>
</body>
</html>

Variables & Data Types

Variables are containers for data values. Declare using var, let, or const.

var a = 2;               // Number
var b = "Reactify Hub";  // String
var c = true;            // Boolean
var d = new Array();     // Array

JavaScript is dynamically typed (type is decided at runtime).

  • Number, String, Boolean, Object, Array, Undefined, Null
var length = 16;                              // Number
var lastName = "Johnson";                     // String
var x = { firstName: "John", lastName: "Doe"}; // Object
var c = true;                                 // Boolean
var d = new Array(1,2,3);                     // Array

Functions

A function is a reusable block of code that can be called from anywhere in the program.

  • Saves time by avoiding repetition.
  • Lets you run code in response to events/actions.
// Declaration
function functionName(optionalParameters){
  // function code
}

// Call
functionName(/* values if any */);

Example: Call Immediately

<html>
<head>
  <script type="text/javascript">
    function f() {
      window.alert("Hello from Reactify Hub");
    }
    f();
  </script>
</head>
<body></body>
</html>

Example: Call via Event

<html>
<head>
  <script type="text/javascript">
    function f() {
      window.alert("Button Clicked....Hello from Reactify Hub");
    }
  </script>
</head>
<body>
  <input type="button" value="Click me" onclick="f()" />
</body>
</html>

Control Statements

1) if / if-else / else-if

<html>
<head>
  <script type="text/javascript">
    function f(){
      var a = 10;
      var b = 18;
      if (a <= 10) {
        document.write("a value is less than or equal to 10<br/>");
      }
      if (b <= 10) {
        document.write("b value is less than or equal to 10<br/>");
      } else if (b < 15) {
        document.write("b value is less than 15<br/>");
      } else {
        document.write("b value is greater than or equal to 15<br/>");
      }
    }
    f();
  </script>
</head>
<body></body>
</html>

2) switch

<html>
<head>
  <script type="text/javascript">
    function f(){
      var a = 17;
      switch(a){
        case 0:  document.write("a is 0"); break;
        case 5:  document.write("a is 5"); break;
        case 10: document.write("a is 10"); break;
        default: document.write("a is not matched with any case");
      }
    }
    f();
  </script>
</head>
<body></body>
</html>

3) while

<html>
<head>
  <script type="text/javascript">
    function f(){
      var a = 0;
      while(a < 10){
        document.write("a is : " + a + "<br/>");
        a++;
      }
    }
    f();
  </script>
</head>
<body></body>
</html>

4) for

<html>
<head>
  <script type="text/javascript">
    function f(){
      for (var a = 0; a < 10; a++){
        document.write("a is : " + a + "<br/>");
      }
    }
    f();
  </script>
</head>
<body></body>
</html>

Objects

Objects are collections of properties (key–value pairs).

var person = {
  name: "Santosh",
  age: 35,
  channel: "Reactify Hub",
  getmydetails: function(){ return "my name is: " + this.name; }
};

document.write("age is " + person.age + "<br/>");
document.write(person.getmydetails());
<html>
<head>
  <script type="text/javascript">
    function f(){
      var person = {name:"Santosh", age:35, channel:"Reactify Hub",
                    getmydetails: function(){ return "my name is : " + this.name; }};
      // add properties dynamically
      person["city"] = "Visakhapatnam";
      document.write("age is " + person.age + "<br/>");
      document.write(person.getmydetails());
    }
    f();
  </script>
</head>
<body></body>
</html>

Error Handling

Errors stop program execution. Use try, catch, throw, and finally to handle them.

Bad Example (will error)

<html>
<head>
  <script type="text/javascript">
    function f(){
      document.write("Before Error Line");
      window.alerting("Hello Welcome"); // typo: alerting
      document.write("After Error Line");
    }
    f();
  </script>
</head>
<body></body>
</html>

Handled with try–catch–finally

<html>
<head>
  <script type="text/javascript">
    function f(){
      document.write("Before Error Line<br/>");
      try{
        window.alerting("Hello Welcome"); // will throw
      } catch(err){
        alert("error is: " + err.message);
      } finally {
        document.write("After Error Line in finally");
      }
    }
    f();
  </script>
</head>
<body></body>
</html>

User-defined Exception

<html>
<head>
  <script type="text/javascript">
    function f(){
      var input = window.prompt("enter a number below 10");
      var num = parseInt(input);
      if (num < 10) document.write("Correct input");
      try{
        if (num >= 10) throw new Error("Wrong input");
      } catch(err){
        alert(err.message);
      }
    }
    f();
  </script>
</head>
<body></body>
</html>
